byte	System.Byte	8 bit	da 0 a 255
sbyte	System.SByte	8 bit	da -128 a 127

ushort	System.UInt16	16 bit	da 0 a 65.535
short	System.Int16	16 bit	da -32.768 a 32.767

uint	System.UInt32	32 bit	da 0 a 4,2 miliardi
int	System.Int32	32 bit	da -2,1 miliardi a +2,1 miliardi

ulong	System.UInt64	64 bit	da 0 a (18000 miliardi di miliardi)
long	System.Int64	64 bit	Molto grande (9000 miliardi di miliardi)

UInt128	System.UInt128	128 bit Estremamente grande
Int128	System.Int128	128 bit	Estremamente grande

nint (System.IntPtr): Intero con segno di dimensione nativa.
nuint (System.UIntPtr): Intero senza segno di dimensione nativa.
Su un sistema a 32 bit, un IntPtr occupa 4 byte (32 bit).
Su un sistema a 64 bit, un IntPtr occupa 8 byte (64 bit).
Integer Pointer - è un riferimento a una risorsa del sistema operativo.
